docker build -t delta-lake .

docker run -it delta-lake


# Kafka commands
docker-compose up

# Connect to any node
docker exec -it kafka1 /bin/bash

# Create a topic
kafka-topics --bootstrap-server kafka1:9092 --create --topic exampleTopic

# Use kafka producer to put message into a topic
kafka-console-producer --bootstrap-server kafka1:9092 --topic exampleTopic
"Hello World!"

# Use kafka consumer to get message from a topic
kafka-console-consumer --bootstrap-server kafka3:9092 --topic exampleTopic --from-beginning
